import pandas as pd

from sklearn.linear_model import LinearRegression


# Carga los datos del archivo CSV

data = pd.read_csv(‘Edgarcore/regresiones/seriesgithub.csv')


# Imprime las columnas disponibles en el archivo CSV

print('Columnas disponibles:')

print(data.columns)


# Solicita al usuario que ingrese la variable dependiente

variable_dependiente = input('Ingrese el nombre de la variable dependiente: ')


# Solicita al usuario que ingrese las variables independientes

variables_independientes = input('Ingrese los nombres de las variables independientes separadas por coma: ')

variables_independientes = [var.strip() for var in variables_independientes.split(',')]


# Verifica que las variables ingresadas sean válidas

if variable_dependiente not in data.columns or any(var not in data.columns for var in variables_independientes):

    print('Error: una o más variables ingresadas no existen en el archivo CSV.')

else:

    # Solicita al usuario que ingrese el periodo de tiempo

    periodo_inicio = input('Ingrese el periodo de inicio (YYYY-Q): ')

    periodo_fin = input('Ingrese el periodo de fin (YYYY-Q): ')


    # Filtra los datos según el periodo de tiempo ingresado

    filtro_periodo = (data['periodo'] >= periodo_inicio) & (data['periodo'] <= periodo_fin)

    data_filtrada = data.loc[filtro_periodo]


    # Define las variables independientes (X) y la variable dependiente (y)

    X = data_filtrada[variables_independientes]

    y = data_filtrada[variable_dependiente]


    # Crea una instancia del modelo de regresión lineal

    regression_model = LinearRegression()


    # Ajusta el modelo utilizando los datos

    regression_model.fit(X, y)


    # Imprime los coeficientes y el término independiente

    print('Coeficientes:', regression_model.coef_)

    print('Término independiente:', regression_model.intercept_)
